%----------------------------------------------------------------------------------------
%	JOB COMMAND
%----------------------------------------------------------------------------------------

\newcommand{\jobheader}[6]{
  \begin{tabular}{l|l}
    \makecell[l]{
      \textbf{#1}
      \\
      to \textbf{#2}
    }
    &
    \makecell[l]{
      \textbf{#3}
      \\
      \textit{#4}
    }
  \end{tabular}


  \textbf{Technologies and skills}{#5}


  {#6}


  \vspace {0.75em}
}


%----------------------------------------------------------------------------------------
%	MY TABBED BLOCK COMMAND
%----------------------------------------------------------------------------------------
\newcommand{\mytabbedblock}[2]{
\begin{tabbing}
\hspace{2.2cm} \= \kill
\textbf{#1} \>
\begin{minipage}{\smallertextwidth}
#2
\end{minipage}
\end{tabbing}
}
